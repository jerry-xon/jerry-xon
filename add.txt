 //swapping
 #include<stdio.h>
 
 main()
 {
    int x,y,t;
    printf("enter two numbers x and y: ");
    scanf("%d %d", &x,&y);
    printf("before swapping x=%d,y=%d",x,y);
    
    t=x;
    x=y;
    y=t;
    printf("after swapping x=%d, y=%d" , x,y);
    return 0;

 }
